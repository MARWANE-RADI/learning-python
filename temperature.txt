
unit = input("is this temperature in celsuis or fahrenheit (C/F) :")

    print(f" The temperature in Fahrenheit is : {temp}°F")
elif unit == "F" :
    temp=round((temp-32)*5/9,1)
    print(f"tempurature in celsuis is: {temp}°C")
